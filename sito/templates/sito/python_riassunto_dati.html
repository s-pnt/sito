# LISTE

numeri = [1, 2, 3]     # dichiarazione lista tramite parentesi quadre
numeri = list((1,2,3)) # dichiarazione lista tramite metodo list() il quale si aspetta una tupla
                       # per questo la necessità di usare le doppie parentesi tonde
numeri = list((1,))    # dichiarazione lista con singolo elemento tramite metodo list()
                       # la virgola dopo l'1 indica a Python che quel dato è una tupla
numeri.append(4)       # metodo append(), aggiunge un elemento alla fine della lista
numeri + [4]           # come il metodo append() sopra
numeri.insert(i, 4)    # metodo insert(), inserisce il valore 4 alla posizione i
numeri.pop(i)          # metodo pop(), restituisce ed elimina l'elemento alla posizione i
numeri.remove(3)       # metodo remove(), rimuove il VALORE 3, non alla posizione 3
numeri.extend([5,6])   # metodo extend(), aggiunge più elementi senza creare una sottolista
numeri.index(5)        # metodo index(), restituisce l'indice del valore 5
